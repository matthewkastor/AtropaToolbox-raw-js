 56 /**
 57  * Pushes a requirement check into atropa.data.requirements. The test
 58  *  tests whether the class is supported in this environment. Sets
 59  *  atropa.data[className]'s support to unsupported and error to errorMessage
 60  *  if the requirementFn returns false. The requirement checks will all be run
 61  *  after the library has loaded.
 62  * @author <a href="mailto:matthewkastor@gmail.com">
 63  *  Matthew Christopher Kastor-Inare III </a><br />
 64  *  ☭ Hial Atropa!! ☭
 65  * @version 20130308
 66  * @param {String} className The name of the class.
 67  * @param {Function} requirementFn A function to test whether or not the class
 68  *  is supported in this environment. If supported, returns true otherwise
 69  *  return false.
 70  * @param {String} errorMessage The error message to use when this class or its
 71  *  methods are called in unsupported environments. Defaults to:
 72  *  'The atropa.' + className + ' class is unsupported in this environment.';
 73  */
 74 atropa.requires = function (className, requirementFn, errorMessage) {
 75     "use strict";
 76     var check = function () {
 77         var test = false;
 78         if(typeof className !== 'string') {
 79             throw new Error('atropa.requires requires the class name to be ' +
 80                 'specified');
 81         }
 82         
 83         if(atropa.data[className] === undefined) {
 84             atropa.data[className] = {};
 85             
 86             if(typeof requirementFn !== 'function') {
 87                 requirementFn = false;
 88             }
 89             
 90             if(typeof errorMessage !== 'string') {
 91                 errorMessage = 'The atropa.' + className +
 92                     ' class is unsupported in this environment.';
 93             }
 94             
 95             try {
 96                 test = requirementFn();
 97             } catch (e) {
 98                 test = false;
 99             }
100             
101             atropa.data[className].error = errorMessage;
102             
103             if(test === false) {
104                 atropa.data[className].support = 'unsupported';
105             }
106         }
107     };
108     
109     atropa.data.requirements.push(check);
110 };

128  * This represents a filter for arguments based on type.
129  * @author <a href="mailto:matthewkastor@gmail.com">

135  * @requires atropa.arrays.match
136  * @see <a href="../../../AtropaToolboxTests.html?spec=atropa.ArgsInfo">tests</a>
137  * @example
138  * function myClassyConstructor(takes, a, few, args) {
139  *     var expectedArgTypes, checker;
140  *     
141  *     expectedArgTypes = {};
142  *     expectedArgTypes.requestWithMessage = 
143  *          ['string', 'string', 'string', 'function'];
144  *     expectedArgTypes.requestNullMessage = 
145  *          ['string', 'string', 'object', 'function'];
146  *     
147  *     checker = new atropa.ArgsInfo();
148  *     checker.setExpectedArgTypes(expectedArgTypes);
149  *     
150  *     try {
151  *     
152  *         // Check the supplied arguments pseudo array's argument types
153  *         // if the pattern of types in arguments matches one of the
154  *         // patterns set on expectedArgTypes then the matching pattern
155  *         // will be returned. Otherwise, an error will be thrown.
156  *         
157  *         checker.checkArgTypes(arguments);
158  *     } catch (e) {
159  *     
160  *         // Invalid argument types supplied. Handle
161  *         // the error or bail.
162  *         
163  *     }
164  *     
165  *     // the arguments supplied will be of the proper type
166  *     // your function can go ahead and do things with them
167  * }
168  */
169 atropa.ArgsInfo = function ArgsInfo() {
170     'use strict';
171     var expectedArgTypes,
172     checkArgs,
173     that;
174     /**
175      * Holds the proper reference to <code>this</code>
176      * for private functions.
177      * @type This
178      * @private
179      * @fieldOf atropa.ArgsInfo-
180      */
181     that = this;
182     /**
183      * Holds the expected argument types object.
184      * @private
185      * @type Expected Arg Types
186      * @fieldOf atropa.ArgsInfo-
187      */
188     expectedArgTypes = {};
189     /**
190      * Sets the expected argument types.
191      * @author <a href="mailto:matthewkastor@gmail.com">
192      *  Matthew Christopher Kastor-Inare III </a><br />
193      *  ☭ Hial Atropa!! ☭
194      * @version 20120909
195      * @methodOf atropa.ArgsInfo#
196      * @param {Expected Arg Types} typesObj An object containing information
197      *  about the types of arguments you expect. Specifically, the object should
198      *  look like the example.
199      * @example
200      * // typesObj is expected to be of the form:
201      * 
202      * var typesObj = {
203      *     "namedArgumentTypesArray" : ["string", "function", "number"],
204      *     "namedAlternateArgumentTypesArray" : ["object", "function", "number"]
205      * };
206      * 
207      * // You may use as many named arrays as you wish and checkArgTypes will
208      * // test for a match to at least one of the provided named arrays.
209      * @throws {atropa.InvalidArgumentTypesError} Throws an error if the
210      *  typesObj can not be used to set the expected argument types.
211      */
212     this.setExpectedArgTypes = function setExpectedArgTypes(typesObj) {
213         var error, names;
214         
215         error = false;
216         
217         if(atropa.inquire.isObjectNotNull(typesObj)) {
218             names = Object.keys(typesObj);
219             if (names.length > 0) {
220                 expectedArgTypes = typesObj;
221             } else {
222                 error = true;
223             }
224         } else {
225             error = true;
226         }
227         
228         if(error) {
229             throw new atropa.InvalidArgumentTypesError(
230                 'typesObj is expected to be of the form: var typesObj = ' +
231                 '{ "namedArgumentTypesArray" : ' +
232                 '    ["string", "function", "number"], ' +
233                 '"namedAlternateArgumentTypesArray" : ' +
234                 '   ["object", "function", "number"] }; ' +
235                 'You may use as many named arrays as you wish and' +
236                 'checkArgTypes will test for a match to at least one of the ' +
237                 'provided named arrays.'
238             );
239         }
240     };
241     /**
242      * Gets the types of arguments.
243      * @author <a href="mailto:matthewkastor@gmail.com">
244      *  Matthew Christopher Kastor-Inare III </a><br />
245      *  ☭ Hial Atropa!! ☭
246      * @version 20120909
247      * @methodOf atropa.ArgsInfo#
248      * @param {arguments} args An arguments object, or anything you want to
249      * check the type of.
250      * @returns {Array} Returns an array of the types of arguments passed in.
251      */
252     this.getArgTypes = function getArgTypes(args) {
253         var x,
254         argTypes;
255         argTypes = [];
256         for (x in args) {
257             if (args.hasOwnProperty(x)) {
258                 argTypes.push(typeof(args[x]));
259             }
260         }
261         return argTypes;
262     };
263     /**
264      * Compares the expected arguments types to the
265      * received arguments types.
266      * @author <a href="mailto:matthewkastor@gmail.com">
267      *  Matthew Christopher Kastor-Inare III </a><br />
268      *  ☭ Hial Atropa!! ☭
269      * @version 20120909
270      * @private
271      * @methodOf atropa.ArgsInfo-
272      * @param {Array} expectedTypesArray An array taken from the user
273      * created argument types object.
274      * @param {arguments} args an arguments object.
275      * @returns {Boolean} Returns true if the expected types match for type
276      *  and are in the same order as the received types.
277      * @requires atropa.arrays.match
278      */
279     checkArgs = function checkArgs(expectedTypesArray, args) {
280         var types;
281         types = {};
282         types.expected = expectedTypesArray;
283         types.received = that.getArgTypes(args);
284         return atropa.arrays.match(types.expected, types.received);
285     };
286     /**
287      * Checks the given arguments object against the expected
288      * arguments types.
289      * @author <a href="mailto:matthewkastor@gmail.com">
290      *  Matthew Christopher Kastor-Inare III </a><br />
291      *  ☭ Hial Atropa!! ☭
292      * @version 20120909
293      * @methodOf atropa.ArgsInfo#
294      * @param {arguments} args An arguments object
295      * @returns {String} The user assigned key which matches the
296      * arguments supplied, or throws an error.
297      * @throws {atropa.InvalidArgumentTypesError} Throws an error if no matching
298      *  pattern of argument types can be found for <code>args</code>
299      * @see atropa.ArgsInfo#setExpectedArgTypes
300      */
301     this.checkArgTypes = function checkArgTypes(args) {
302         var expectedTypes;
303         if (Object.keys(expectedArgTypes).length < 1) {
304             throw new atropa.InvalidArgumentTypesError(
305                 'Expected argument types is not set. Use ' +
306                 'setExpectedArgTypes(typesObj) to set. typesObj is an ' +
307                 'object whose properties are arrays of strings representing ' +
308                 'the typeof(argument) for each argument, in the exact order ' +
309                 'in which they will be given to the function. Using multiple ' +
310                 'properties it is possible to define alternative acceptable ' +
311                 'argument type sets. Use getArgTypes(arguments) as a ' +
312                 'convenient way of getting the array you want to hard code ' +
313                 'in for validation. Example: var typesObj = ' +
314                 '{ "messageIncluded" : ["string", "function", "number"], ' +
315                 '"messageNotIncluded" : ["object", "function", "number"] };'
316             );
317         }
318         for (expectedTypes in expectedArgTypes) {
319             if (expectedArgTypes.hasOwnProperty(expectedTypes)) {
320                 if (checkArgs(expectedArgTypes[expectedTypes], args)) {
321                     return expectedTypes;
322                 }
323             }
324         }
325         throw new atropa.InvalidArgumentTypesError(
326             'invalid argument type @ atropa.ArgsInfo.checkArgTypes');
327     };
328 };

342 /**
343  * Compares two arrays based on size, contents, and element order.
344  * @author <a href="mailto:matthewkastor@gmail.com">
345  *  Matthew Christopher Kastor-Inare III </a><br />
346  *  ☭ Hial Atropa!! ☭
347  * @version 20120909
348  * @param {Array} array1 One array you want compared to another.
349  * @param {Array} array2 The other array.
350  * @returns {Boolean} Returns true or false depending on
351  *  whether or not the arrays matched in size, composition, and
352  *  element order.
353  * @example
354  * var x = [1,2];
355  * var y = [1,1,3];
356  * atropa.arrays.match(x,y);
357  * // returns false
358  * @example
359  * var x = [1,2];
360  * var y = [1,2];
361  * atropa.arrays.match(x,y);
362  * // returns true
363  * @example
364  * var x = [1,2];
365  * var y = [2,1];
366  * atropa.arrays.match(x,y);
367  * // returns false because the elements are not in the same order.
368  * @example
369  * var x = [1,{'aProp' : 'aValue'}];
370  * var y = [1,{'aProp' : 'aValue'}];
371  * atropa.arrays.match(x,y);
372  * // returns false because even though the object looks the same, the
373  * // two objects are in fact distinct objects.
374  * @example
375  * var obj = {'aProp' : 'aValue'};
376  * var x = [1,obj];
377  * var y = [1,obj];
378  * atropa.arrays.match(x,y);
379  * // returns true because the objects referenced in the arrays are
380  * // in fact the same object.
381  */
382 atropa.arrays.match = function arraysMatch(array1, array2) {
383     "use strict";
384     var x,
385     l;
386     if (array1.length !== array2.length) {
387         return false;
388     }
389     l = array1.length;
390     for (x = 0; x < l; x += 1) {
391         if (array1[x] !== array2[x]) {
392             return false;
393         }
394     }
395     return true;
396 };

499 atropa.arrays.intersect = function intersect(array1, array2) {
500     "use strict";
501     var smallArray, largeArray, intersection = [];
502     if(array1.length > array2.length) {
503         largeArray = array1.splice(0);
504         smallArray = array2.splice(0);
505     } else {
506         largeArray = array2.splice(0);
507         smallArray = array1.splice(0);
508     }
509     smallArray.forEach(function (item) {
510         var idxInLargeArray = largeArray.indexOf(item);
511         if (0 <= idxInLargeArray) { // has word
512             intersection.push(largeArray.splice(idxInLargeArray, 1)[0]);
513         }
514     });
515     return intersection;
516 };

613 /**

642  * @author <a href="mailto:matthewkastor@gmail.com">
643  *  Matthew Christopher Kastor-Inare III </a><br />
644  *  ☭ Hial Atropa!! ☭
645  * @version 20130118
646  * @param {Array} arr The array with discontinuous keys.
647  * @returns {Array} Returns an array with continuous keys.
648  * @example
649  * var x = [ "a", "b", "c", undefined ];
650  * console.log(x); // [ "a", "b", "c", undefined ]
651  * console.log(x.length); // 4
652  * 
653  * delete x[1]; // deletes the key from the array but
654  *              // the array length remains the same
655  *              // at this point the arrays keys are 0, 2, and 3
656  * console.log(x); // [ "a", undefined, "c", undefined ]
657  * console.log(x.length); // 4
658  * 
659  * x = atropa.arrays.reindex(x);
660  * console.log(x); //  [ "a", "c", undefined ]
661  *    // note that the last element existed in the array, its value was
662  *    // undefined but it did have a key so the element remains in the array.
663  *    //
664  *    // The deleted element was in fact deleted from the array so there was no
665  *    // key x[1] at all, when trying to access this non existing element the

674     var idx, out;
675     out = [];
676     for(idx in arr) {
677         if(arr.hasOwnProperty(idx)) {
678             out.push(arr[idx]);
679         }
680     }
681     return out;
682 };
