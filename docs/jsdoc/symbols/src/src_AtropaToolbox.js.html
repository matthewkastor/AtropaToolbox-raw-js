 33 /**
 34  * This represents a filter for arguments based on type.
 35  * @author <a href="mailto:matthewkastor@gmail.com">
 36  *  Matthew Christopher Kastor-Inare III </a><br />
 37  *  ☭ Hial Atropa!! ☭
 38  * @version 20130221
 39  * @class This represents a filter for arguments based on type.
 40  * @returns {ArgsInfo} Returns an ArgsInfo filter.
 41  * @requires atropa.arrays.match
 42  * @example
 43  * function myClassyConstructor(takes, a, few, args) {
 44  *     var expectedArgTypes, checker;
 45  *     
 46  *     expectedArgTypes = {};
 47  *     expectedArgTypes.requestWithMessage = ['string', 'string', 'string', 'function'];
 48  *     expectedArgTypes.requestNullMessage = ['string', 'string', 'object', 'function'];
 49  *     
 50  *     checker = new atropa.ArgsInfo();
 51  *     checker.setExpectedArgTypes(expectedArgTypes);
 52  *     
 53  *     try {
 54  *     
 55  *         // Check the supplied arguments pseudo array's argument types
 56  *         // if the pattern of types in arguments matches one of the
 57  *         // patterns set on expectedArgTypes then the matching pattern
 58  *         // will be returned. Otherwise, an error will be thrown.
 59  *         
 60  *         checker.checkArgTypes(arguments);
 61  *     } catch (e) {
 62  *     
 63  *         // Invalid argument types supplied. Handle
 64  *         // the error or bail.
 65  *         
 66  *     }
 67  *     
 68  *     // the arguments supplied will be of the proper type
 69  *     // your function can go ahead and do things with them
 70  * }
 71  */
 72 atropa.ArgsInfo = function ArgsInfo() {
 73     'use strict';
 74     var expectedArgTypes,
 75     checkArgs,
 76     that;
 77     /**
 78      * Holds the proper reference to <code>this</code>
 79      * for private functions.
 80      * @type This
 81      * @private
 82      * @fieldOf atropa.ArgsInfo-
 83      */
 84     that = this;
 85     /**
 86      * Holds the expected argument types object.
 87      * @private
 88      * @type Expected Arg Types
 89      * @fieldOf atropa.ArgsInfo-
 90      */
 91     expectedArgTypes = {};
 92     /**
 93      * Sets the expected argument types.
 94      * @author <a href="mailto:matthewkastor@gmail.com">
 95      *  Matthew Christopher Kastor-Inare III </a><br />
 96      *  ☭ Hial Atropa!! ☭
 97      * @version 20120909
 98      * @methodOf atropa.ArgsInfo#
 99      * @param {Expected Arg Types} typesObj An object containing information about the
100      * types of arguments you expect. Specifically, the object should look like the example.
101      * @example
102      * // typesObj is expected to be of the form:
103      * 
104      * var typesObj = {
105      *     "namedArgumentTypesArray" : ["string", "function", "number"],
106      *     "namedAlternateArgumentTypesArray" : ["object", "function", "number"]
107      * };
108      * 
109      * // You may use as many named arrays as you wish and checkArgTypes will
110      * // test for a match to at least one of the provided named arrays.
111      * @throws {atropa.InvalidArgumentTypesError} Throws an error if the typesObj
112      *  can not be used to set the expected argument types.
113      */
114     this.setExpectedArgTypes = function setExpectedArgTypes(typesObj) {
115         var names;
116         names = Object.keys(typesObj);
117         if (names.length < 1) {
118             throw new atropa.InvalidArgumentTypesError('typesObj is expected to be of the form: var typesObj = ' +
119                 '{ "namedArgumentTypesArray" : ["string", "function", "number"]' +
120                 ', "namedAlternateArgumentTypesArray" : ["object", "function",' +
121                 '"number"] }; You may use as many named arrays as you wish and' +
122                 'checkArgTypes will test for a match to at least one of the ' +
123                 'provided named arrays.');
124         }
125         expectedArgTypes = typesObj;
126     };
127     /**
128      * Gets the types of arguments.
129      * @author <a href="mailto:matthewkastor@gmail.com">
130      *  Matthew Christopher Kastor-Inare III </a><br />
131      *  ☭ Hial Atropa!! ☭
132      * @version 20120909
133      * @methodOf atropa.ArgsInfo#
134      * @param {arguments} args An arguments object, or anything you want to
135      * check the type of.
136      * @returns {Array} Returns an array of the types of arguments passed in.
137      */
138     this.getArgTypes = function getArgTypes(args) {
139         var x,
140         argTypes;
141         argTypes = [];
142         for (x in args) {
143             if (args.hasOwnProperty(x)) {
144                 argTypes.push(typeof(args[x]));
145             }
146         }
147         return argTypes;
148     };
149     /**
150      * Compares the expected arguments types to the
151      * received arguments types.
152      * @author <a href="mailto:matthewkastor@gmail.com">
153      *  Matthew Christopher Kastor-Inare III </a><br />
154      *  ☭ Hial Atropa!! ☭
155      * @version 20120909
156      * @private
157      * @methodOf atropa.ArgsInfo-
158      * @param {Array} expectedTypesArray An array taken from the user
159      * created argument types object.
160      * @param {arguments} args an arguments object.
161      * @returns {Boolean} Returns true if the expected types match for type
162      * and in the same order as the received types.
163      * @requires atropa.arrays.match
164      */
165     checkArgs = function checkArgs(expectedTypesArray, args) {
166         var types;
167         types = {};
168         types.expected = expectedTypesArray;
169         types.received = that.getArgTypes(args);
170         return atropa.arrays.match(types.expected, types.received);
171     };
172     /**
173      * Checks the given arguments object against the expected
174      * arguments types.
175      * @author <a href="mailto:matthewkastor@gmail.com">
176      *  Matthew Christopher Kastor-Inare III </a><br />
177      *  ☭ Hial Atropa!! ☭
178      * @version 20120909
179      * @methodOf atropa.ArgsInfo#
180      * @param {arguments} args An arguments object
181      * @returns {String} The user assigned key which matches the
182      * arguments supplied, or throws an error.
183      * @throws {atropa.InvalidArgumentTypesError} Throws an error if no matching
184      *  pattern of argument types can be found for <code>args</code>
185      * @see atropa.ArgsInfo#setExpectedArgTypes
186      */
187     this.checkArgTypes = function checkArgTypes(args) {
188         var expectedTypes;
189         if (Object.keys(expectedArgTypes).length < 1) {
190             throw new atropa.InvalidArgumentTypesError('Expected argument types is not set. Use ' +
191                 'setExpectedArgTypes(typesObj) to set. typesObj is an ' +
192                 'object whose properties are arrays of strings representing ' +
193                 'the typeof(argument) for each argument, in the exact order ' +
194                 'in which they will be given to the function. Using multiple ' +
195                 'properties it is possible to define alternative acceptable ' +
196                 'argument type sets. Use getArgTypes(arguments) as a ' +
197                 'convenient way of getting the array you want to hard code ' +
198                 'in for validation. Example: var typesObj = ' +
199                 '{ "messageIncluded" : ["string", "function", "number"], ' +
200                 '"messageNotIncluded" : ["object", "function", "number"] };');
201         }
202         for (expectedTypes in expectedArgTypes) {
203             if (expectedArgTypes.hasOwnProperty(expectedTypes)) {
204                 if (checkArgs(expectedArgTypes[expectedTypes], args)) {
205                     return expectedTypes;
206                 }
207             }
208         }
209         throw new atropa.InvalidArgumentTypesError('invalid argument type @ atropa.ArgsInfo.checkArgTypes');
210     };
211 };

227 /**
228  * Compares two arrays based on size, contents, and element order.
229  * @author <a href="mailto:matthewkastor@gmail.com">
230  *  Matthew Christopher Kastor-Inare III </a><br />
231  *  ☭ Hial Atropa!! ☭
232  * @version 20120909
233  * @param {Array} array1 One array you want compared to another.
234  * @param {Array} array2 The other array.
235  * @returns {Boolean} Returns true or false depending on
236  *  whether or not the arrays matched in size, composition, and
237  *  element order.
238  * @example
239  * var x = [1,2];
240  * var y = [1,1,3];
241  * atropa.arrays.match(x,y);
242  * // returns false
243  * @example
244  * var x = [1,2];
245  * var y = [1,2];
246  * atropa.arrays.match(x,y);
247  * // returns true
248  * @example
249  * var x = [1,2];
250  * var y = [2,1];
251  * atropa.arrays.match(x,y);
252  * // returns false because the elements are not in the same order.
253  * @example
254  * var x = [1,{'aProp' : 'aValue'}];
255  * var y = [1,{'aProp' : 'aValue'}];
256  * atropa.arrays.match(x,y);
257  * // returns false because even though the object looks the same, the
258  * // two objects are in fact distinct objects.
259  * @example
260  * var obj = {'aProp' : 'aValue'};
261  * var x = [1,obj];
262  * var y = [1,obj];
263  * atropa.arrays.match(x,y);
264  * // returns true because the objects referenced in the arrays are
265  * // in fact the same object.
266  */
267 atropa.arrays.match = function arraysMatch(array1, array2) {
268     "use strict";
269     var x,
270     l;
271     if (array1.length !== array2.length) {
272         return false;
273     }
274     l = array1.length;
275     for (x = 0; x < l; x += 1) {
276         if (array1[x] !== array2[x]) {
277             return false;
278         }
279     }
280     return true;
281 };
282 /**
283  * Subtracts one array from another array based on the unique values in both sets.
284  * @author <a href="mailto:matthewkastor@gmail.com">
285  *  Matthew Christopher Kastor-Inare III </a><br />
286  *  ☭ Hial Atropa!! ☭
287  * @version 20130112
288  * @param {Array} a The array to subtract.
289  * @param {Array} fromB The array with elements duplicated in <code>a</code>
290  * @returns {Array} Returns a new array containing only the unique
291  *  values found in <code>fromB</code> that are not present in <code>a</code>
292  * @example
293  * var x = [1,2];
294  * var y = [1,1,3];
295  * atropa.arrays.subtract(x,y);
296  * // returns [3]
297  * @example
298  * var x = [1,3];
299  * var y = [3,1];
300  * atropa.arrays.subtract(x,y);
301  * // returns []
302  * @example
303  * var x = [1,3];
304  * var y = [3,1,1,9];
305  * atropa.arrays.subtract(x,y);
306  * // returns [9]
307  * @example
308  * var x = [1,3,{'aProp' : 'aVal'}];
309  * var y = [3,1,{'aProp' : 'aVal'}];
310  * atropa.arrays.subtract(x,y);
311  * // returns [{'aProp' : 'aVal'}] because the two objects are not the same object.
312  * @example
313  * var obj = {'aProp' : 'aVal'};
314  * var x = [1,3,obj];
315  * var y = [3,1,{'aProp' : 'aVal'}];
316  * atropa.arrays.subtract(x,y);
317  * // returns [{'aProp' : 'aVal'}] because the two objects are not the same object.
318  * @example
319  * var obj = {'aProp' : 'aVal'}
320  * var x = [1,3,obj];
321  * var y = [3,1,obj];
322  * atropa.arrays.subtract(x,y);
323  * // returns [] because the objects referenced in the arrays are the same object.
324  */
325 atropa.arrays.subtract = function(a, fromB) {
326     "use strict";
327     var the = {};
328     the.result = [];
329     fromB.forEach(function(item){
330         the.mark = false;
331         a.forEach(function(rm){
332             if(item === rm) {
333                 the.mark = true;
334             }
335         });
336         if(the.mark !== true) {
337             the.result.push(item);
338         }
339     });
340     return the.result;
341 };
342 /**
343  * Returns an array of values found in both of the given arrays.
344  * @author <a href="mailto:matthewkastor@gmail.com">
345  *  Matthew Christopher Kastor-Inare III </a><br />
346  *  ☭ Hial Atropa!! ☭
347  * @version 20130112
348  * @param {Array} array1 An array.
349  * @param {Array} array2 Another array.
350  * @returns {Array} Returns an array of values found in both of the given arrays.
351  * @example
352  * var x = [1,3,4];
353  * var y = [3,1,5];
354  * atropa.arrays.intersect(x,y);
355  * // returns [1,3]
356  * @example
357  * var x = [1,1,3,4];
358  * var y = [3,1,1,5];
359  * atropa.arrays.intersect(x,y);
360  * // returns [1,1,3]
361  * @example
362  * var obj = {'aProp' : 'aVal'};
363  * var x = [1,3,obj];
364  * var y = [3,1,obj];
365  * atropa.arrays.intersect(x,y);
366  * // returns [1,3,{'aProp' : 'aVal'}]
367  * @example
368  * var obj = {'aProp' : 'aVal'};
369  * var x = [1,3,{'aProp' : 'aVal'}];
370  * var y = [3,1,obj];
371  * atropa.arrays.intersect(x,y);
372  * // returns [1,3] because the two objects are not the same object.
373  * @example
374  * var x = [1,3,{'aProp' : 'aVal'}];
375  * var y = [3,1,{'aProp' : 'aVal'}];
376  * atropa.arrays.intersect(x,y);
377  * // returns [1,3] because the two objects are not the same object.
378  */
379 atropa.arrays.intersect = function intersect(array1, array2) {
380     "use strict";
381     var smallArray, largeArray, intersection = [];
382     if(array1.length > array2.length) {
383         largeArray = array1.splice(0);
384         smallArray = array2.splice(0);
385     } else {
386         largeArray = array2.splice(0);
387         smallArray = array1.splice(0);
388     }
389     smallArray.forEach(function (item, idx, arr) {
390         var idxInLargeArray = largeArray.indexOf(item);
391         if (0 <= idxInLargeArray) { // has word
392             intersection.push(largeArray.splice(idxInLargeArray, 1)[0]);
393         }
394     });
395     return intersection;
396 };
397 /**
398  * Calculates the frequency of items occurring in an array.
399  * @author <a href="mailto:matthewkastor@gmail.com">
400  *  Matthew Christopher Kastor-Inare III </a><br />
401  *  ☭ Hial Atropa!! ☭
402  * @version 20130118
403  * @param {Array} arr The array to calculate frequencies from.
404  * @returns {Object} Returns an object whose keys are each unique
405  *  elements from the array and their value is their frequency of
406  *  occurrence within the array. Be careful that your array does
407  *  not contain values matching object instance property names.
408  * @example
409  * var x = [1,1,1,1,1,3,3];
410  * atropa.arrays.getFrequency(x);
411  * // returns {
412  * //     "1": 5,
413  * //     "3": 2
414  * // }
415  * @example
416  * var x = ["bill", "fred", "fred", "jane"];
417  * atropa.arrays.getFrequency(x);
418  * // returns {
419  * //     "bill": 1,
420  * //     "fred": 2,
421  * //     "jane": 1
422  * // }
423  * @example
424  * var x = [1,3,{'aProp' : 'aVal'}];
425  * atropa.arrays.getFrequency(x);
426  * // returns {
427  * //     "1": 1,
428  * //     "3": 1,
429  * //     "[object Object]": 1
430  * // }
431  * @example
432  * var obj = {'aProp' : 'aVal'};
433  * var otherObj = {};
434  * var x = [1,3,obj,otherObj,{'aDoughnut' : 'sprinkles'}];
435  * atropa.arrays.getFrequency(x);
436  * // returns {
437  * //     "1": 1,
438  * //     "3": 1,
439  * //     "[object Object]": 3
440  * // }
441  * @example
442  * var x = [1,3,"toString"];
443  * atropa.arrays.getFrequency(x);
444  * // returns {
445  * //     "1": 1,
446  * //     "3": 1,
447  * //     "toString": "function toString() {\n    [native code]\n}1"
448  * // }
449  */
450 atropa.arrays.getFrequency = function (arr) {
451     "use strict";
452     var out = arr.reduce(function (acc, curr) {
453         if (acc[curr] === undefined) {
454             acc[curr] = 1;
455         } else {
456             acc[curr] += 1;
457         }
458         return acc;
459     }, {});
460     return out;
461 };

515 /**
516  * Reindexes an array.
517  * @author <a href="mailto:matthewkastor@gmail.com">
518  *  Matthew Christopher Kastor-Inare III </a><br />
519  *  ☭ Hial Atropa!! ☭
520  * @version 20130118
521  * @param {Array} arr The array with discontinuous keys.
522  * @returns {Array} Returns an array with continuous keys.
523  * @example
524  * var x = [ "a", "b", "c", undefined ];
525  * console.log(x); // [ "a", "b", "c", undefined ]
526  * console.log(x.length); // 4
527  * 
528  * delete x[1]; // deletes the key from the array but
529  *              // the array length remains the same
530  *              // at this point the arrays keys are 0, 2, and 3
531  * console.log(x); // [ "a", undefined, "c", undefined ]
532  * console.log(x.length); // 4
533  * 
534  * x = atropa.arrays.reindex(x);
535  * console.log(x); //  [ "a", "c", undefined ]
536  *    // note that the last element existed in the array,
537  *    // its value was undefined but it did have a key so
538  *    // the element remains in the array.
539  *    //
540  *    // The deleted element was in fact deleted from the array so there was no
541  *    // key x[1] at all, when trying to access this non existing element the
542  *    // value of undefined was returned. 
543  *    // This behavior is confusing unless you
544  *    // think about the arrayas an object whose properties are named by numbers.
545  *    // Accessing an undefined property returns undefined regardless of
546  *    // whether the property existed in the past or not.
547  * console.log(x.length); // 3
548  */
549 atropa.arrays.reindex = function reindex(arr) {
550     "use strict";
551     var idx, out;
552     out = [];
553     for(idx in arr) {
554         if(arr.hasOwnProperty(idx)) {
555             out.push(arr[idx]);
556         }
557     }
558     return out;
559 };

604 /**
605  * A polling class designed for executing long running processes that return nothing and have no callback parameter.
606  * @class A polling class designed for executing long running processes that return nothing and have no callback parameter.
607  * @author <a href="mailto:matthewkastor@gmail.com">
608  *  Matthew Christopher Kastor-Inare III </a><br />
609  *  ☭ Hial Atropa!! ☭
610  * @version 20130220
611  * @param {String} actorName The name for the SerialActor instance.
612  * @param {Function} actorFunction The function to execute when the
613  *  SerialActor is free. This function must call the <code>free</code> function
614  *  when it is finished in order to allow the actor to continue.
615  * @returns {atropa.SerialActor} Returns an <code>atropa.SerialActor</code>
616  *  instance.
617  * @example
618  * function dummyActor(){
619  *     var that = this;
620  *     console.log('actorFunction would execute');
621  *     console.log('freeing ' + this.name + ' in 10000 ms');
622  *     setTimeout(function(){that.free();}, 10000);
623  * };
624  * var actor = new atropa.SerialActor('dummy', dummyActor);
625  *     // change the name of the actor from
626  *     // dummy to awesome
627  * actor.name = "awesome";
628  *     // set the polling interval (milliseconds)
629  * actor.interval = 3000;
630  *     // set the blocking timeout value (milliseconds)
631  * actor.blockTimeoutValue = 120000;
632  *     // start polling
633  * actor.start();
634  *     // dynamically change the SerialActor
635  * setTimeout(function(){
636  *     // change the polling interval
637  *     // while the SerialActor is running.
638  *     actor.changeInterval(2000);
639  *         // change the actor function
640  *     actor.actorFunction = function() {
641  *         console.log('new actorFunction executing');
642  *         console.log('freeing ' + this.name + ' immediately');
643  *         this.free();
644  *     };
645  * },10000);
646  */
647 atropa.SerialActor = function(actorName, actorFunction) {
648     "use strict";
649     var that, dummyActor;
650     /**
651      * Reference to <code>this</code>
652      * @fieldOf atropa.SerialActor-
653      * @private
654      * @type {Object}
655      */
656     that = this;
657     /**
658      * Default actorFunction
659      * @author <a href="mailto:matthewkastor@gmail.com">
660      *  Matthew Christopher Kastor-Inare III </a><br />
661      *  ☭ Hial Atropa!! ☭
662      * @version 20130220
663      * @methodOf atropa.SerialActor-
664      * @private
665      * @see atropa.SerialActor#actorFunction
666      * @example
667      * dummyActor = function(){
668      *     console.log('actorFunction would execute');
669      *     console.log('freeing Serial Actor in 10000 ms');
670      *     setTimeout(function(){that.free();}, 10000);
671      * };
672      */
673     dummyActor = function(){
674         console.log('actorFunction would execute');
675         console.log('freeing Serial Actor in 10000 ms');
676         setTimeout(function(){that.free();}, 10000);
677     };
678     /**
679      * The name of this instance. Defaults to "SerialActor"
680      * @fieldOf atropa.SerialActor#
681      * @type String
682      * @default "SerialActor"
683      */
684     this.name = atropa.setAsOptionalArg('SerialActor', actorName);
685     /**
686      * Polling interval in milliseconds. This determines how frequently the actor function will
687      *  try to execute. Defaults to 100 milliseconds.
688      * @fieldOf atropa.SerialActor#
689      * @type Number
690      * @default 100
691      */
692     this.interval = 100; // milliseconds
693     /**
694      * The id of the interval set to poll the actor. You should not change
695      *  this manually, use the start and stop functions instead. Defauls to undefined.
696      * @fieldOf atropa.SerialActor#
697      * @type Number
698      * @default undefined
699      */
700     this.intervalId = undefined;
701     /**
702      * The state of the SerialActor. If true, the actor will sleep. If false the actor
703      *  will execute the actor function when next polled. Defaults to false.
704      * @fieldOf atropa.SerialActor#
705      * @type Boolean
706      * @default false
707      */
708     this.blocked = false;
709     /**
710      * Stores id's of currently running timeout functions used to free the actor
711      *  if it has been blocked for too long.
712      * @fieldOf atropa.SerialActor#
713      * @see atropa.SerialActor#blockTimeoutValue
714      * @type Array
715      * @default []
716      */
717     this.timeouts = [];
718     /**
719      * The maximum time, in milliseconds, which the actor may be blocked for.
720      *  After this duration has been reached the actor will be freed. Defaults to 60 seconds.
721      * @fieldOf atropa.SerialActor#
722      * @type Number
723      * @default 60000
724      */
725     this.blockTimeoutValue = 60000;
726     /**
727      * The function to execute when the SerialActor is free. This function
728      *  must call the <code>free</code> function when it is finished in order to
729      *  allow the actor to continue. Defaults to the <code>dummyActor</code>
730      *  function.
731      * @fieldOf atropa.SerialActor#
732      * @type Function
733      * @default dummyActor
734      * @see atropa.SerialActor-dummyActor
735      * @example
736      * dummyActor = function(){
737      *     console.log('actorFunction would execute');
738      *     console.log('freeing Serial Actor in 10000 ms');
739      *     setTimeout(function(){that.free();}, 10000);
740      * };
741      */
742     this.actorFunction = atropa.setAsOptionalArg(dummyActor, actorFunction);
743     /**
744      * The action function is called when the actor is polled and it's blocked state is false.
745      *  This method should not be set or called manually, set the <code>actorFunction</code> instead.
746      * @author <a href="mailto:matthewkastor@gmail.com">
747      *  Matthew Christopher Kastor-Inare III </a><br />
748      *  ☭ Hial Atropa!! ☭
749      * @version 20130220
750      * @methodOf atropa.SerialActor#
751      * @see atropa.SerialActor#actorFunction
752      */
753     this.action = function() {
754         if(false === that.blocked) {
755             that.block();
756             setTimeout(function() {
757                 that.actorFunction();
758             }, 100);
759         } else {
760             console.log(that.name + ' sleeping for ' + that.interval + ' ms');
761         }
762     };
763 };
764 /**
765  * Prevents the actor from executing it's actorFunction.
766  *  This block will timeout once the <code>blockTimeoutValue</code> has been reached.
767  * @author <a href="mailto:matthewkastor@gmail.com">
768  *  Matthew Christopher Kastor-Inare III </a><br />
769  *  ☭ Hial Atropa!! ☭
770  * @version 20130220
771  * @methodOf atropa.SerialActor#
772  * @returns {Boolean} Returns the value of this instances <code>blocked</code> property.
773  * @see atropa.SerialActor#blocked
774  * @example
775  * function d() {
776  *     console.log('doing stuff to things');
777  *     this.free();
778  * }
779  * 
780  * var actor = new atropa.SerialActor('dummy', d);
781  * actor.interval = 2000;
782  * actor.blockTimeoutValue = 5000;
783  * actor.start();
784  * // 5 seconds after starting the actor will be blocked.
785  * // It will remain blocked until the block timeout is reached.
786  * setTimeout(function() {
787  *     console.log('blocking!!!');
788  *     actor.block();
789  * }, 5000);
790  */
791 atropa.SerialActor.prototype.block = function() {
792     "use strict";
793     var that = this;
794     console.log(this.name + ' block');
795     this.blocked = true;
796     this.timeouts.push(setTimeout(function() {that.blockTimeout();}, that.blockTimeoutValue));
797     return this.blocked;
798 };
799 /**
800  * Called when the <code>blockTimeoutValue</code> has been reached. This frees the actor
801  *  and removes the timeout reference from the timeouts array.
802  * @author <a href="mailto:matthewkastor@gmail.com">
803  *  Matthew Christopher Kastor-Inare III </a><br />
804  *  ☭ Hial Atropa!! ☭
805  * @version 20130220
806  * @methodOf atropa.SerialActor#
807  * @returns {Boolean} Returns the value of this instances <code>blocked</code> property.
808  * @see atropa.SerialActor#blocked
809  */
810 atropa.SerialActor.prototype.blockTimeout = function() {
811     "use strict";
812     console.log(this.name + ' block timeout');
813     return this.free();
814 };
815 /**
816  * Frees the actor so it may execute its actor function when next polled.
817  * @author <a href="mailto:matthewkastor@gmail.com">
818  *  Matthew Christopher Kastor-Inare III </a><br />
819  *  ☭ Hial Atropa!! ☭
820  * @version 20130220
821  * @methodOf atropa.SerialActor#
822  * @returns {Boolean} Returns the value of this instances <code>blocked</code> property.
823  * @see atropa.SerialActor#blocked
824  * @example
825  * function d() {
826  *     console.log('doing stuff to things');
827  *     this.free();
828  * }
829  * 
830  * var actor = new atropa.SerialActor('dummy', d);

840 atropa.SerialActor.prototype.free = function() {
841     "use strict";
842     console.log(this.name + ' free');
843     this.blocked = false;
844     clearTimeout(this.timeouts.shift());
845     return this.blocked;
846 };

849  * @author <a href="mailto:matthewkastor@gmail.com">

854  * @param {Number} interval Optional. The polling interval. Defaults to the value

856  * @see atropa.SerialActor#interval
857  * @returns {Number} Returns the value of this instance's <code>intervalId</code> property.
858  * @see atropa.SerialActor#intervalId
859  * @example
860  * var actor = new atropa.SerialActor('dummy');

863 atropa.SerialActor.prototype.start = function(interval) {
864     "use strict";
865     var that = this;
866     this.interval = atropa.setAsOptionalArg(this.interval, interval);
867     
868     if(this.intervalId !== undefined) {
869         // clear the old timeout before creating a new one.
870         this.stop();
871     }
872     this.intervalId = setInterval(that.action, that.interval);
873     console.log(this.name + ' started');
874     return this.intervalId;
875 };

895 atropa.SerialActor.prototype.changeInterval = function(interval) {
896     "use strict";

898     return this.start(interval);
899 };

930 /**
931  * This class represents a babbler. The babbler
932  * produces lorum ipsum text, to user specifications.
933  * @author <a href="mailto:matthewkastor@gmail.com">
934  *  Matthew Christopher Kastor-Inare III </a><br />
935  *  ☭ Hial Atropa!! ☭
936  * @version 20130223
937  * @class This class represents a babbler
938  * @param {Number} wrdCount The amount of "words" you would like
939  * the babbler to produce.
940  * @returns {Babbler} Returns a babbler.
941  * @requires atropa.random.integer
942  * @requires atropa.string.ucFirst
943  * @requires atropa.random.string
944  * @example
945  * var babbler = new atropa.Babbler(30);
946  * // resets the word count
947  * babbler.resetWordCount(10)
948  * console.log(babbler.getWordCount());
949  * 
950  * // displays a 10 word sentence of nonsense words.
951  * console.log(babbler.generateBabble(10));
952  * // displays a 3 word sentence
953  * console.log(babbler.generateBabble(3));
954  * 
955  * // displays the user stored or last generated babble
956  * console.log(babbler.getBabble());
957  * 
958  * // clears the stored babble
959  * babbler.resetBabble();
960  * console.log(babbler.getBabble());
961  * 
962  * // sets the babble
963  * babbler.setBabble('here be gibberish ');
964  * console.log(babbler.getBabble());
965  * 
966  * // append more gibberish to the current babble
967  * babbler.setBabble(babbler.getBabble() + babbler.generateBabble(5));
968  * console.log(babbler.getBabble());
969  * 
970  * // generate a sentence
971  * babbler.resetWordCount(10);
972  * console.log(babbler.generateSentence(5, 20));
973  * 
974  * // generate random punctuation
975  * console.log(babbler.punctuate());
976  * 
977  * // generate a word
978  * console.log(babbler.generateWord(3,7));
979  * console.log(babbler.generateWord(7,10));
980  */
981 atropa.Babbler = function Babbler(wrdCount) {
982     'use strict';
983     var babble = '',
984     wordCount = 0;
985     /**
986      * Sets the word count.
987      * @author <a href="mailto:matthewkastor@gmail.com">
988      *  Matthew Christopher Kastor-Inare III </a><br />
989      *  ☭ Hial Atropa!! ☭
990      * @version 20120909
991      * @methodOf atropa.Babbler#
992      * @param {Number} wrdCount The amount of "words" which you want the
993      * babbler to produce.
994      * @returns {Number} Returns the set word count for this babbler.
995      */
996     this.setWordCount = function (wrdCount) {
997         if (typeof wrdCount !== 'number') {
998             if (typeof wordCount !== 'number') {
999                 wordCount = 250;
1000             }
1001         } else {
1002             wordCount = wrdCount;
1003         }
1004         return wordCount;
1005     };
1006     /**
1007      * Resets the word count for this babbler.
1008      * @author <a href="mailto:matthewkastor@gmail.com">
1009      *  Matthew Christopher Kastor-Inare III </a><br />
1010      *  ☭ Hial Atropa!! ☭
1011      * @version 20120909
1012      * @methodOf atropa.Babbler#
1013      * @param {Number} wordCount The amount of "words" you would like
1014      * to set for this babbler.
1015      * @returns {Number} Returns the set word count for this babbler.
1016      */
1017     this.resetWordCount = function resetWordCount(wordCount) {
1018         this.setWordCount(wordCount);
1019         return wordCount;
1020     };
1021     /**
1022      * Gets the current word count.
1023      * @author <a href="mailto:matthewkastor@gmail.com">
1024      *  Matthew Christopher Kastor-Inare III </a><br />
1025      *  ☭ Hial Atropa!! ☭
1026      * @version 20120909
1027      * @methodOf atropa.Babbler#
1028      * @returns {Number} Returns the word count for this babbler.
1029      */
1030     this.getWordCount = function getWordCount() {
1031         return wordCount;
1032     };
1033     /**
1034      * Generates a word with a specified length. Lowers the word count by one.
1035      * @author <a href="mailto:matthewkastor@gmail.com">
1036      *  Matthew Christopher Kastor-Inare III </a><br />
1037      *  ☭ Hial Atropa!! ☭
1038      * @version 20120909
1039      * @methodOf atropa.Babbler#
1040      * @param {Number} stringMin the shortest word, in characters.
1041      * @param {Number} stringMax The longest word, in characters.
1042      * @returns {String} Returns a random string of characters
1043      * within the specified range of length.
1044      * @requires atropa.random.integer
1045      * @requires atropa.random.string
1046      */
1047     this.generateWord = function generateWord(stringMin, stringMax) {
1048         var wordLength,
1049         word;
1050         wordLength = atropa.random.integer(stringMin, stringMax);
1051         word = atropa.random.string(wordLength, 'lower');
1052         wordCount--;
1053         return word;
1054     };
1055     /**
1056      * Provides random punctuation.
1057      * @author <a href="mailto:matthewkastor@gmail.com">
1058      *  Matthew Christopher Kastor-Inare III </a><br />
1059      *  ☭ Hial Atropa!! ☭
1060      * @version 20120909
1061      * @methodOf atropa.Babbler#
1062      * @returns {String} Returns a random punctuation
1063      * character ( . ! or ? ).
1064      * @requires atropa.random.string
1065      */
1066     this.punctuate = function punctuate() {
1067         var punctuation;
1068         punctuation = atropa.random.string(1, 'punctuation');
1069         return punctuation;
1070     };
1071     /**
1072      * Generates a sentence of specified length in words. The quantity
1073      *  of words in the generated sentence will be between the minimum
1074      *  and maximum set, with the maximum capped at the current words
1075      *  count. The word count will be lowered by the
1076      *  quantity of words in the generated sentence. If the word count
1077      *  is 0 then there will be no words in the sentence. If the word
1078      *  count is 3 then the maximum possible number of words in the
1079      *  sentence will be three.
1080      * @author <a href="mailto:matthewkastor@gmail.com">
1081      *  Matthew Christopher Kastor-Inare III </a><br />
1082      *  ☭ Hial Atropa!! ☭
1083      * @version 20120909
1084      * @methodOf atropa.Babbler#
1085      * @param {Number} sentenceMin The shortest sentence, in words,
1086      * you would like returned.
1087      * @param {Number} sentenceMax The longest sentence, in words,
1088      * you would like returned.
1089      * @returns {String} Returns a "sentence" within the specified
1090      * range of length.
1091      * @requires atropa.random.integer
1092      * @requires atropa.string.ucFirst
1093      */
1094     this.generateSentence = function generateSentence(sentenceMin, sentenceMax) {
1095         var word,
1096         sentenceLength,
1097         sentence;
1098         sentenceLength = atropa.random.integer(sentenceMin, sentenceMax);
1099         sentence = '';
1100         if (sentenceLength > wordCount) {
1101             sentenceLength = wordCount;
1102         }
1103         for (sentenceLength; sentenceLength > 0; sentenceLength--) {
1104             if (wordCount > 0) {
1105                 word = this.generateWord(4, 12);
1106                 sentence += ' ' + word;
1107             } else {
1108                 sentenceLength = 0;
1109             }
1110         }
1111         sentence += this.punctuate();
1112         return atropa.string.ucFirst(sentence);
1113     };
1114     /**
1115      * Sets the babble.
1116      * @author <a href="mailto:matthewkastor@gmail.com">
1117      *  Matthew Christopher Kastor-Inare III </a><br />
1118      *  ☭ Hial Atropa!! ☭
1119      * @version 20120909
1120      * @methodOf atropa.Babbler#
1121      * @param {String} babbleString Specified babble to set.
1122      * @returns {String} Returns the stored babble.
1123      */
1124     this.setBabble = function setBabble(babbleString) {
1125         if (typeof babbleString === 'string') {
1126             babble = babbleString;
1127         } else {
1128             this.resetBabble();
1129         }
1130         return babble;
1131     };
1132     /**
1133      * Clears the stored babble.
1134      * @author <a href="mailto:matthewkastor@gmail.com">
1135      *  Matthew Christopher Kastor-Inare III </a><br />
1136      *  ☭ Hial Atropa!! ☭
1137      * @version 20120909
1138      * @methodOf atropa.Babbler#
1139      * @returns {String} Returns the stored babble.
1140      */
1141     this.resetBabble = function resetBabble() {
1142         babble = '';
1143         return babble;
1144     };
1145     /**
1146      * Gets the last generated babble.
1147      * @author <a href="mailto:matthewkastor@gmail.com">
1148      *  Matthew Christopher Kastor-Inare III </a><br />
1149      *  ☭ Hial Atropa!! ☭
1150      * @version 20120909
1151      * @methodOf atropa.Babbler#
1152      * @returns {String} Returns the stored babble.
1153      */
1154     this.getBabble = function getBabble() {
1155         return babble;
1156     };
1157     /**
1158      * Generates babble to a user specified length in words.
1159      *  The word count will be zero after this and the stored
1160      *  babble will be set to the generated babble.
1161      * @author <a href="mailto:matthewkastor@gmail.com">
1162      *  Matthew Christopher Kastor-Inare III </a><br />
1163      *  ☭ Hial Atropa!! ☭
1164      * @version 20120909
1165      * @methodOf atropa.Babbler#
1166      * @param {Number} wordsCt The desired word count for the
1167      * generated babble.
1168      * @returns {String} Returns babble of specified length in words.
1169      * @see atropa.Babbler#getWordCount
1170      */
1171     this.generateBabble = function generateBabble(wordsCt) {
1172         this.resetBabble();
1173         this.resetWordCount(wordsCt);
1174         for (wordCount; wordCount > 0; babble += ' ') {
1175             this.setBabble(babble + this.generateSentence(5, 20));
1176         }
1177         return babble;
1178     };
1179     this.resetWordCount(wrdCount);
1180     return this;
1181 };
