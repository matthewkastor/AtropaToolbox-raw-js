 28 /**
 29  * This represents a filter for arguments based on type.
 30  * @author <a href="mailto:matthewkastor@gmail.com">
 31  * Matthew Christopher Kastor-Inare III </a><br />
 32  * ☭ Hial Atropa!! ☭
 33  * @version 20120909
 34  * @class This represents a filter for arguments based on type.
 35  * @returns {ArgsInfo} Returns an ArgsInfo filter.
 36  * @requires atropa.arrays.match
 37  */
 38 atropa.ArgsInfo = function ArgsInfo() {
 39 	'use strict';
 40 	var expectedArgTypes,
 41 	checkArgs,
 42 	that;
 43 	/**
 44 	 * Holds the proper reference to <code>this</code>
 45 	 * for private functions.
 46 	 * @type This
 47 	 * @private
 48 	 * @fieldOf atropa.ArgsInfo-
 49 	 */
 50 	that = this;
 51 	/**
 52 	 * Holds the expected argument types object.
 53 	 * @private
 54 	 * @type Expected Arg Types
 55 	 * @fieldOf atropa.ArgsInfo-
 56 	 */
 57 	expectedArgTypes = {};
 58 	/**
 59 	 * Sets the expected argument types.
 60 	 * @author <a href="mailto:matthewkastor@gmail.com">
 61 	 * Matthew Christopher Kastor-Inare III </a><br />
 62 	 * ☭ Hial Atropa!! ☭
 63 	 * @version 20120909
 64 	 * @methodOf atropa.ArgsInfo#
 65 	 * @param {Expected Arg Types} typesObj An object containing information about the
 66 	 * types of arguments you expect. Specifically, the object should look like the example.
 67 	 * @example
 68 	 * <p><b> * typesObj is expected to be of the form: </b></p>
 69 	 * <pre>
 70 	 * var typesObj = {
 71 	 *     "namedArgumentTypesArray" :
 72 	 *     ["string", "function", "number"],
 73 	 *     "namedAlternateArgumentTypesArray" :
 74 	 *     ["object", "function", "number"]
 75 	 * };
 76 	 * </pre>
 77 	 * <p>You may use as many named arrays as you wish and checkArgTypes will
 78 	 * test for a match to <u>at least one</u> of the provided named arrays.</p>
 79 	 * @returns {undefined}
 80 	 * @throws invalid argument type.
 81 	 */
 82 	this.setExpectedArgTypes = function setExpectedArgTypes(typesObj) {
 83 		var names;
 84 		names = Object.keys(typesObj);
 85 		if (names.length < 1) {
 86 			throw 'typesObj is expected to be of the form: var typesObj = '
 87 			 + '{ "namedArgumentTypesArray" : ["string", "function", "number"]'
 88 			 + ', "namedAlternateArgumentTypesArray" : ["object", "function",'
 89 			 + '"number"] }; You may use as many named arrays as you wish and'
 90 			 + 'checkArgTypes will test for a match to at least one of the '
 91 			 + 'provided named arrays.';
 92 		}
 93 		expectedArgTypes = typesObj;
 94 	};
 95 	/**
 96 	 * Gets the types of arguments.
 97 	 * @author <a href="mailto:matthewkastor@gmail.com">
 98 	 * Matthew Christopher Kastor-Inare III </a><br />
 99 	 * ☭ Hial Atropa!! ☭
100 	 * @version 20120909
101 	 * @methodOf atropa.ArgsInfo#
102 	 * @param {arguments} args An arguments object, or anything you want to
103 	 * check the type of.
104 	 * @returns {Array} Returns an array of the types of arguments passed in.
105 	 */
106 	this.getArgTypes = function getArgTypes(args) {
107 		var x,
108 		argTypes;
109 		argTypes = [];
110 		for (x in args) {
111 			if (args.hasOwnProperty(x)) {
112 				argTypes.push(typeof(args[x]));
113 			}
114 		}
115 		return argTypes;
116 	};
117 	/**
118 	 * Compares the expected arguments types to the
119 	 * received arguments types.
120 	 * @author <a href="mailto:matthewkastor@gmail.com">
121 	 * Matthew Christopher Kastor-Inare III </a><br />
122 	 * ☭ Hial Atropa!! ☭
123 	 * @version 20120909
124 	 * @private
125 	 * @methodOf atropa.ArgsInfo-
126 	 * @param {Array} expectedTypesArray An array taken from the user
127 	 * created argument types object.
128 	 * @param {arguments} args an arguments object.
129 	 * @returns {Boolean} Returns true if the expected types match for type
130 	 * and in the same order as the received types.
131 	 * @requires atropa.arrays.match
132 	 */
133 	checkArgs = function checkArgs(expectedTypesArray, args) {
134 		var types;
135 		types = {};
136 		types.expected = expectedTypesArray;
137 		types.received = that.getArgTypes(args);
138 		return atropa.arrays.match(types.expected, types.received);
139 	};
140 	/**
141 	 * Checks the given arguments object against the expected
142 	 * arguments types.
143 	 * @author <a href="mailto:matthewkastor@gmail.com">
144 	 * Matthew Christopher Kastor-Inare III </a><br />
145 	 * ☭ Hial Atropa!! ☭
146 	 * @version 20120909
147 	 * @methodOf atropa.ArgsInfo#
148 	 * @param {arguments} args An arguments object
149 	 * @returns {String} The user assigned key which matches the
150 	 * arguments supplied, or throws an error.
151 	 * @throws invalid argument type
152 	 */
153 	this.checkArgTypes = function checkArgTypes(args) {
154 		var expectedTypes;
155 		if (Object.keys(expectedArgTypes).length < 1) {
156 			throw 'Expected argument types is not set. Use '
157 			 + 'setExpectedArgTypes(typesObj) to set. typesObj is an '
158 			 + 'object whose properties are arrays of strings representing '
159 			 + 'the typeof(argument) for each argument, in the exact order '
160 			 + 'in which they will be given to the function. Using multiple '
161 			 + 'properties it is possible to define alternative acceptable '
162 			 + 'argument type sets. Use getArgTypes(arguments) as a '
163 			 + 'convenient way of getting the array you want to hard code '
164 			 + 'in for validation. Example: var typesObj = '
165 			 + '{ "messageIncluded" : ["string", "function", "number"], '
166 			 + '"messageNotIncluded" : ["object", "function", "number"] };';
167 		}
168 		for (expectedTypes in expectedArgTypes) {
169 			if (expectedArgTypes.hasOwnProperty(expectedTypes)) {
170 				if (checkArgs(expectedArgTypes[expectedTypes], args)) {
171 					return expectedTypes;
172 				}
173 			}
174 		}
175 		throw 'invalid argument type @ checkArgTypes';
176 	};
177 };

192 /**
193  * Compares two arrays based on size and contents.
194  * @author <a href="mailto:matthewkastor@gmail.com">
195  * Matthew Christopher Kastor-Inare III </a><br />
196  * ☭ Hial Atropa!! ☭
197  * @version 20120909
198  * @param {Array} array1 One array you want compared to another.
199  * @param {Array} array2 The other array.
200  * @returns {Boolean} Returns true or false depending on
201  * whether or not the arrays matched in size and composition.
202  */
203 atropa.arrays.match = function arraysMatch(array1, array2) {
204 	"use strict";
205 	var x,
206 	l;
207 	if (array1.length !== array2.length) {
208 		return false;
209 	}
210 	l = array1.length;
211 	for (x = 0; x < l; x += 1) {
212 		if (array1[x] !== array2[x]) {
213 			return false;
214 		}
215 	}
216 	return true;
217 };
218 /**
219  * Subtracts one array of scalar values from another array of scalar values.
220  * @author <a href="mailto:matthewkastor@gmail.com">
221  * Matthew Christopher Kastor-Inare III </a><br />
222  * ☭ Hial Atropa!! ☭
223  * @version 20130112
224  * @param {Array} a The array to subtract.
225  * @param {Array} fromB The array with elements duplicated in <code>a</code>
226  * @example
227  * <pre>
228  * var x = [1,2];
229  * var y = [1,3];
230  * atropa.arrays.subtract(x,y);
231  * // [3]
232  * </pre>
233  * @returns {Array} Returns a new array containing only the unique
234  *  values found in the largeArray.
235  */
236 atropa.arrays.subtract = function(a, fromB) {
237 	"use strict";
238     var the = {};
239     the.result = [];
240     fromB.forEach(function(item){
241         the.mark = false;
242         a.forEach(function(rm){
243             if(item === rm) {
244                 the.mark = true;
245             }
246         });
247         if(the.mark !== true) {
248             the.result.push(item);
249         }
250     });
251     return the.result;
252 };
253 /**
254  * Calculates the frequency of items occurring in an array.
255  * @author <a href="mailto:matthewkastor@gmail.com">
256  * Matthew Christopher Kastor-Inare III </a><br />
257  * ☭ Hial Atropa!! ☭
258  * @version 20130118
259  * @param {Array} arr The array to calculate frequencies from.
260  * @returns {Object} Returns an object whose keys are each unique
261  *  elements from the array and their value is their frequency of
262  *  occurrence within the array.
263  */
264 atropa.arrays.getFrequency = function (arr) {
265 	"use strict";
266 	var out = arr.reduce(function (acc, curr) {
267 		if (acc[curr] === undefined) {
268 			acc[curr] = 1;
269 		} else {
270 			acc[curr] += 1;
271 		}
272 		return acc;
273 	}, {});
274 	return out;
275 };

310 /**
311  * This class represents a babbler. The babbler
312  * produces lorum ipsum text, to user specifications.
313  * @author <a href="mailto:matthewkastor@gmail.com">
314  * Matthew Christopher Kastor-Inare III </a><br />
315  * ☭ Hial Atropa!! ☭
316  * @version 20120909
317  * @class This class represents a babbler
318  * @param {Number} wrdCount The amount of "words" you would like
319  * the babbler to produce.
320  * @returns {Babbler} Returns a babbler.
321  * @requires atropa.random.integer
322  * @requires atropa.string.ucFirst
323  * @requires atropa.random.string
324  */
325 atropa.Babbler = function Babbler(wrdCount) {
326 	'use strict';
327 	var babble = '',
328 	wordCount = 0;
329 	/**
330 	 * Sets the word count.
331 	 * @author <a href="mailto:matthewkastor@gmail.com">
332 	 * Matthew Christopher Kastor-Inare III </a><br />
333 	 * ☭ Hial Atropa!! ☭
334 	 * @version 20120909
335 	 * @methodOf atropa.Babbler#
336 	 * @param {Number} wrdCount The amount of "words" which you want the
337 	 * babbler to produce.
338 	 * @returns {Number} Returns the set word count for this babbler.
339 	 */
340 	this.setWordCount = function (wrdCount) {
341 		if (typeof(wrdCount) !== 'number') {
342 			if (typeof(wordCount) !== 'number') {
343 				wordCount = 250;
344 			}
345 		} else {
346 			wordCount = wrdCount;
347 		}
348 		return wordCount;
349 	};
350 	/**
351 	 * Resets the word count for this babbler.
352 	 * @author <a href="mailto:matthewkastor@gmail.com">
353 	 * Matthew Christopher Kastor-Inare III </a><br />
354 	 * ☭ Hial Atropa!! ☭
355 	 * @version 20120909
356 	 * @methodOf atropa.Babbler#
357 	 * @param {Number} wordCount The amount of "words" you would like
358 	 * to set for this babbler.
359 	 * @returns {Number} Returns the set word count for this babbler.
360 	 */
361 	this.resetWordCount = function resetWordCount(wordCount) {
362 		this.setWordCount(wordCount);
363 		return wordCount;
364 	};
365 	/**
366 	 * Gets the current word count.
367 	 * @author <a href="mailto:matthewkastor@gmail.com">
368 	 * Matthew Christopher Kastor-Inare III </a><br />
369 	 * ☭ Hial Atropa!! ☭
370 	 * @version 20120909
371 	 * @methodOf atropa.Babbler#
372 	 * @returns {Number} Returns the word count for this babbler.
373 	 */
374 	this.getWordCount = function getWordCount() {
375 		return wordCount;
376 	};
377 	/**
378 	 * Generates a word with a specified length.
379 	 * @author <a href="mailto:matthewkastor@gmail.com">
380 	 * Matthew Christopher Kastor-Inare III </a><br />
381 	 * ☭ Hial Atropa!! ☭
382 	 * @version 20120909
383 	 * @methodOf atropa.Babbler#
384 	 * @param {Number} stringMin the shortest word, in characters.
385 	 * @param {Number} stringMax The longest word, in characters.
386 	 * @returns {String} Returns a random string of characters
387 	 * within the specified range of length.
388 	 * @requires atropa.random.integer
389 	 * @requires atropa.random.string
390 	 */
391 	this.generateWord = function generateWord(stringMin, stringMax) {
392 		var wordLength,
393 		word;
394 		wordLength = atropa.random.integer(stringMin, stringMax);
395 		word = atropa.random.string(wordLength, 'lower');
396 		wordCount--;
397 		return word;
398 	};
399 	/**
400 	 * Adds punctuation to the babble.
401 	 * @author <a href="mailto:matthewkastor@gmail.com">
402 	 * Matthew Christopher Kastor-Inare III </a><br />
403 	 * ☭ Hial Atropa!! ☭
404 	 * @version 20120909
405 	 * @methodOf atropa.Babbler#
406 	 * @returns {String} Returns a random punctuation
407 	 * character ( . ! or ? ).
408 	 * @requires atropa.random.string
409 	 */
410 	this.punctuate = function punctuate() {
411 		var punctuation;
412 		punctuation = atropa.random.string(1, 'punctuation');
413 		return punctuation;
414 	};
415 	/**
416 	 * Generates a sentence of specified length in words.
417 	 * @author <a href="mailto:matthewkastor@gmail.com">
418 	 * Matthew Christopher Kastor-Inare III </a><br />
419 	 * ☭ Hial Atropa!! ☭
420 	 * @version 20120909
421 	 * @methodOf atropa.Babbler#
422 	 * @param {Number} sentenceMin The shortest sentence, in words,
423 	 * you would like returned.
424 	 * @param {Number} sentenceMax The longest sentence, in words,
425 	 * you would like returned.
426 	 * @returns {String} Returns a "sentence" within the specified
427 	 * range of length.
428 	 * @requires atropa.random.integer
429 	 * @requires atropa.string.ucFirst
430 	 */
431 	this.generateSentence = function generateSentence(sentenceMin, sentenceMax) {
432 		var word,
433 		sentenceLength,
434 		sentence;
435 		sentenceLength = atropa.random.integer(sentenceMin, sentenceMax);
436 		sentence = this.generateWord(4, 12);
437 		if (sentenceLength > wordCount) {
438 			sentenceLength = wordCount;
439 		}
440 		for (sentenceLength; sentenceLength > 0; sentenceLength--) {
441 			if (wordCount > 0) {
442 				word = this.generateWord(4, 12);
443 				sentence += ' ' + word;
444 			} else {
445 				sentenceLength = 0;
446 			}
447 		}
448 		sentence += this.punctuate();
449 		return atropa.string.ucFirst(sentence);
450 	};
451 	/**
452 	 * Sets the babble.
453 	 * @author <a href="mailto:matthewkastor@gmail.com">
454 	 * Matthew Christopher Kastor-Inare III </a><br />
455 	 * ☭ Hial Atropa!! ☭
456 	 * @version 20120909
457 	 * @methodOf atropa.Babbler#
458 	 * @param {String} babbleString Specified babble to set.
459 	 * @returns {String} Returns Babble.
460 	 */
461 	this.setBabble = function setBabble(babbleString) {
462 		if (typeof(babbleString) === 'string') {
463 			babble = babbleString;
464 		} else {
465 			this.resetBabble();
466 		}
467 		return babble;
468 	};
469 	/**
470 	 * Resets the babble.
471 	 * @author <a href="mailto:matthewkastor@gmail.com">
472 	 * Matthew Christopher Kastor-Inare III </a><br />
473 	 * ☭ Hial Atropa!! ☭
474 	 * @version 20120909
475 	 * @methodOf atropa.Babbler#
476 	 * @returns {String} Returns an empty string.
477 	 */
478 	this.resetBabble = function resetBabble() {
479 		babble = '';
480 		return babble;
481 	};
482 	/**
483 	 * Gets the babble.
484 	 * @author <a href="mailto:matthewkastor@gmail.com">
485 	 * Matthew Christopher Kastor-Inare III </a><br />
486 	 * ☭ Hial Atropa!! ☭
487 	 * @version 20120909
488 	 * @methodOf atropa.Babbler#
489 	 * @returns {String} Returns Babble.
490 	 */
491 	this.getBabble = function getBabble() {
492 		return babble;
493 	};
494 	/**
495 	 * Generates babble to a user specified length in words.
496 	 * @author <a href="mailto:matthewkastor@gmail.com">
497 	 * Matthew Christopher Kastor-Inare III </a><br />
498 	 * ☭ Hial Atropa!! ☭
499 	 * @version 20120909
500 	 * @methodOf atropa.Babbler#
501 	 * @param {Number} wordsCt The desired word count for the
502 	 * generated babble.
503 	 * @returns {String} Returns babble of specified length in words.
504 	 */
505 	this.generateBabble = function generateBabble(wordsCt) {
506 		this.resetBabble();
507 		this.resetWordCount(wordsCt);
508 		for (wordCount; wordCount > 0; babble += ' ') {
509 			this.setBabble(babble + this.generateSentence(5, 20));
510 		}
511 		return babble;
512 	};
513 	
514 	this.resetBabble();
515 	this.resetWordCount(wrdCount);
516 };

532  * @class This represents a cookie handler
533  * @returns {CookieMonster} A cookie handler.
534  */
535 atropa.CookieMonster = function CookieMonster() {
536 	'use strict';
537 	var currentCookies,
538 	getCookieCallback;
539 	
540 	/**
541 	 * This holds the current cookie object array.
542 	 * @private
543 	 * @type Array
544 	 * @fieldOf atropa.CookieMonster-
545 	 */
546 	currentCookies = [];
547 	/**
548 	 * Converts a cookie string into an object.
549 	 * @author <a href="mailto:matthewkastor@gmail.com">
550 	 * Matthew Christopher Kastor-Inare III </a><br />
551 	 * ☭ Hial Atropa!! ☭
552 	 * @version 20120909
553 	 * @methodOf atropa.CookieMonster#
554 	 * @param {String} cookie A cookie represented as a string
555 	 * <code>cookieName=cookieVal;</code>
556 	 * @returns {cookieObj} Returns a cookie object.
557 	 */
558 	this.cookie2obj = function cookie2obj(cookie) {
559 		var cookieObj = {};
560 		if (!cookie) {
561 			return false;
562 		}
563 		cookieObj.key = cookie.substr(0, cookie.indexOf("="));
564 		cookieObj.val = cookie.substr(cookie.indexOf("=") + 1);
565 		return cookieObj;
566 	};
567 	/**
568 	 * Converts a cookie object to a cookie string.
569 	 * @author <a href="mailto:matthewkastor@gmail.com">
570 	 * Matthew Christopher Kastor-Inare III </a><br />
571 	 * ☭ Hial Atropa!! ☭
572 	 * @version 20120909
573 	 * @methodOf atropa.CookieMonster#
574 	 * @param {Object} cookieObj A cookie object
575 	 * @returns {String} Returns a cookie string.
576 	 */
577 	this.bakeCookie = function bakeCookie(cookieObj) {
578 		var cookie = '',
579 		key,
580 		val;
581 		key = cookieObj.key;
582 		val = cookieObj.val;
583 		cookie = key + '=' + val + ';';
584 		return cookie;
585 	};
586 	/**
587 	 * Checks cookies for worms based on a user defined
588 	 * callback function.
589 	 * @author <a href="mailto:matthewkastor@gmail.com">
590 	 * Matthew Christopher Kastor-Inare III </a><br />
591 	 * ☭ Hial Atropa!! ☭
592 	 * @version 20120909
593 	 * @methodOf atropa.CookieMonster#
594 	 * @param {function} callback
595 	 * @param {Array} args arguments to pass to the callback
596 	 * function
597 	 * @returns {Array} An array of cookie objects.
598 	 */
599 	this.inspectCookies = function inspectCookies(callback, args) {
600 		var testCookie,
601 		cookies,
602 		jar = [];
603 		cookies = this.getCookies();
604 		while (cookies.length > 0) {
605 			testCookie = cookies.shift();
606 			if (callback(testCookie, args) !== false) {
607 				jar.push(testCookie);
608 			}
609 		}
610 		return jar;
611 	};
612 	/**
613 	 * Internal callback function used while getting the current
614 	 * cookies.
615 	 * @author <a href="mailto:matthewkastor@gmail.com">
616 	 * Matthew Christopher Kastor-Inare III </a><br />
617 	 * ☭ Hial Atropa!! ☭
618 	 * @version 20120909
